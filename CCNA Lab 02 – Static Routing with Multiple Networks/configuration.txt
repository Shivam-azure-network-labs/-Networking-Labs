=========================
Router R1 Configuration
=========================

enable
configure terminal

hostname R1

interface FastEthernet0/0
 ip address 192.168.10.1 255.255.255.0
 no shutdown

interface Serial0/0/0
 ip address 10.10.10.1 255.255.255.252
 clock rate 64000
 no shutdown

exit

ip route 192.168.20.0 255.255.255.0 10.10.10.2

end
write memory
