# CCNA Enterprise Mega Lab

## Full Device Configuration

Topology Type: Hybrid Routing (OSPF + Static + Default Route)

---

# ================================

# ISP ROUTER

# ================================

hostname ISP

interface Serial0/0/0
description LINK_TO_HEADQUARTER
ip address 200.0.0.1 255.255.255.252
no shutdown

! Route towards enterprise network
ip route 10.0.0.0 255.0.0.0 200.0.0.2

! Default route simulation
ip route 0.0.0.0 0.0.0.0 200.0.0.2

---

# ================================

# HEADQUARTER ROUTER

# ================================

hostname HEAD-QUARTER

interface GigabitEthernet0/0
description LINK_TO_CORE_A
ip address 10.0.1.1 255.255.255.252
no shutdown

interface GigabitEthernet0/1
description LINK_TO_CORE_B
ip address 10.0.2.1 255.255.255.252
no shutdown

interface Serial0/0/0
description LINK_TO_ISP
ip address 200.0.0.2 255.255.255.252
no shutdown

interface Loopback0
ip address 3.3.3.3 255.255.255.255

! Default route to ISP
ip route 0.0.0.0 0.0.0.0 200.0.0.1

router ospf 1
router-id 3.3.3.3
network 10.0.1.0 0.0.0.3 area 0
network 10.0.2.0 0.0.0.3 area 0
default-information originate

# CORE-A ROUTER

# ================================

hostname CORE-A

description LINK_TO_HEADQUARTER

no shutdown

interface GigabitEthernet0/1
description LINK_TO_SALES
ip address 10.0.3.1 255.255.255.252
no shutdown

interface Loopback0
ip address 1.1.1.1 255.255.255.255

! Static routes to branch networks
ip route 10.0.4.0 255.255.255.252 10.0.3.2
ip route 10.0.5.0 255.255.255.252 10.0.3.2

ip route 192.168.10.0 255.255.255.0 10.0.3.2
ip route 192.168.20.0 255.255.255.0 10.0.3.2
ip route 192.168.30.0 255.255.255.0 10.0.3.2

router-id 1.1.1.1
network 10.0.1.0 0.0.0.3 area 0
network 10.0.3.0 0.0.0.3 area 0

redistribute static subnets
