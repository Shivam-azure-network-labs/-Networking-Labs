==============================
R1 ISP Router
==============================

enable
conf t
hostname ISP

interface s0/0/0
ip address 10.0.12.1 255.255.255.0
no shutdown

interface s0/0/1
ip address 10.0.14.1 255.255.255.0
no shutdown

! Primary Routes

ip route 192.168.10.0 255.255.255.0 10.0.12.2
ip route 192.168.20.0 255.255.255.0 10.0.12.2
ip route 192.168.30.0 255.255.255.0 10.0.14.2

! Backup Routes

ip route 192.168.10.0 255.255.255.0 10.0.14.2 10
ip route 192.168.30.0 255.255.255.0 10.0.12.2 10

end
wr


==============================
R2 HR Router
==============================

enable
conf t
hostname R2

interface g0/0
ip address 10.0.24.1 255.255.255.0
no shutdown

interface g0/1
ip address 192.168.10.1 255.255.255.0
no shutdown

interface s0/0/0
ip address 10.0.12.2 255.255.255.0
no shutdown

! Primary Route

ip route 192.168.30.0 255.255.255.0 10.0.12.1

! Backup Route

ip route 192.168.30.0 255.255.255.0 10.0.24.2 10

! Route to SALES

ip route 192.168.20.0 255.255.255.0 10.0.24.2

end
wr
