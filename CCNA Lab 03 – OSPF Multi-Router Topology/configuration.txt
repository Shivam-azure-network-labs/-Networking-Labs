=================================
R1 Configuration
=================================

enable
configure terminal
hostname R1

interface g0/0
ip address 192.168.10.1 255.255.255.0
no shutdown

interface s0/0/0
ip address 10.10.10.1 255.255.255.252
clock rate 64000
no shutdown

router ospf 1
network 192.168.10.0 0.0.0.255 area 0
network 10.10.10.0 0.0.0.3 area 0

end
write memory
